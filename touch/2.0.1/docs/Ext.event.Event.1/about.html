/*global Ext:false */
Ext.application({
    launch: function() {
        Ext.Viewport.add({
            layout: 'fit',
            items: [{
                docked: 'top',
                xtype: 'toolbar',
                title: 'Ext.event.Event example!'
            }, {
                id: 'logger',
                styleHtmlContent: true,
                html: 'Tap somewhere!',
                padding: 5
            }]
        });

        Ext.Viewport.element.on({
            tap: function(e, node) {
                var string = '';

                string += 'You tapped at: { x: ' + e.pageX + ', y: ' + e.pageY + ' } (e.pageX & e.pageY)';
                string += '
';
                string += 'The HTMLElement you tapped has the className of: ' + e.target.className + ' (e.target)';
                string += '
';
                string += 'The HTMLElement which has the listener has a className of: ' + e.getTarget().className + ' (e.getTarget())';

                Ext.getCmp('logger').setHtml(string);
            }
        });
    }
});
