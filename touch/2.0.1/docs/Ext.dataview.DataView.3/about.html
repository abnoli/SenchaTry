/*global Ext:false */
Ext.application({
    launch: function() {
        Ext.create('Ext.DataView', {
            fullscreen: true,
            cls: 'twitterView',
            store: {
                autoLoad: true,
                fields: ['from_user', 'text', 'profile_image_url'],

                proxy: {
                    type: 'jsonp',
                    url: 'http://search.twitter.com/search.json?q=Sencha Touch',

                    reader: {
                        type: 'json',
                        rootProperty: 'results'
                    }
                }
            },

            itemTpl: '
{from_user}
{text}
'
        });
    }
});
