Specifying which side of the viewport an Ext.Msg component animates from in Sencha Touch 2

The following example shows how you can specify which side of the viewport an Ext.Msg component animates from during a slide animation in Sencha Touch 2 by specifying the enter and exit config options to "left", "right", "top", or "bottom".

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.1
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.Panel', {
            fullscreen: true,
            items: [{
                xtype: 'titlebar',
                docked: 'top',
                items: [{
                    text: 'enter/exit:left',
                    handler: function () {
                        showAlert('left', 'left');
                    } // handler
                }, {
                    text: 'enter/exit:right',
                    handler: function () {
                        showAlert('right', 'right');
                    } // handler
                }, {
                    text: 'enter/exit:top',
                    handler: function () {
                        showAlert('top', 'top');
                    } // handler
                }, {
                    text: 'enter/exit:bottom',
                    handler: function () {
                        showAlert('bottom', 'bottom');
                    } // handler
                }] // items
            }] // items
        }); // create()
        
        function showAlert(enterPos, exitPos) {
            Ext.Msg.show({
                title: 'enter: ' + enterPos + ", exit: " + exitPos,
                showAnimation: 'slideIn',
                hideAnimation: 'slideOut',
                enter: enterPos,
                exit: exitPos
            }); // show()
        } // showAlert()
    } // launch
}); // application()
