Truncating strings in an XTemplate in Sencha Touch 2

The following example shows how you an truncate strings in an Ext.XTemplate in Sencha Touch 2 by using the ellipsis() method.

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.1
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        var myStore = Ext.create('Ext.data.Store', {
            data: [{
                fname: 'Stratton',
                lname: 'Sclavos',
                role: 'Executive Chairman'
            }, {
                fname: 'Michael',
                lname: 'Mullany',
                role: 'CEO'
            }, {
                fname: 'Ted',
                lname: 'Driscoll',
                role: 'Vice President Worldwide Sales'
            }, {
                fname: 'Abraham',
                lname: 'Elias',
                role: 'Chief Technical Officer'
            }, {
                fname: 'Jeff',
                lname: 'Hartley',
                role: 'Vice President of Services and Training'
            }, {
                fname: 'Adam',
                lname: 'Mishcon',
                role: 'Vice President of Operations'
            }, {
                fname: 'Judy',
                lname: 'Lin',
                role: 'Vice President of Engineering'
            }], // data
            sorters: 'lname'
        }); // create()
        
        Ext.create('Ext.List', {
            fullscreen: true,
            store: myStore,
            itemTpl: '
{lname}, {fname} &mdash; {role:ellipsis(22, true)}
',
            items: {
                xtype: 'titlebar',
                title: 'Ext.XTemplate ellipsis()',
                docked: 'top'
            } // items
        }); // create()
    } // launch
}); // application()
