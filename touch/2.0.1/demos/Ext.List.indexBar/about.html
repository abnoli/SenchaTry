/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.List', {
            fullscreen: true,
            indexBar: true,
            grouped: true,
            store: {
                fields: ['name', 'league', 'division'],
                sorters: 'name',
                grouper: {
                    groupFn: function (item) {
                        return item.get('name')[0];
                    } // groupFn
                }, // grouper
