Disabling the Ext.field.Slider control in Sencha Touch 2

The following example shows how you can disable an Ext.field.Slider control in Sencha Touch 2 by setting the Boolean disabled config option or by calling the setDisabled() method.

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.form.Panel', {
            fullscreen: true,
            items: [{
                xtype: 'fieldset',
                items: [{
                    xtype: 'sliderfield',
                    id: 'sl',
                    label: 'slider:',
                    minValue: 0,
                    maxValue: 200,
                    disabled: true
                }, {
                    xtype: 'checkboxfield',
                    id: 'debug',
                    label: 'disabled:',
                    checked: true,
                    listeners: {
                        check: function (ch, evt) {
                            Ext.getCmp('sl').setDisabled(ch.getChecked());
                        }, // check
                        uncheck: function (ch, evt) {
                            Ext.getCmp('sl').setDisabled(ch.getChecked());
                        } // uncheck
                    } // listeners
                }] // items (fieldset)
            }] // items (panel)
        }); // create()
    } // launch
}); // application()
