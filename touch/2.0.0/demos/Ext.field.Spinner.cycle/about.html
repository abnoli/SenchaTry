Toggling value wrapping on an Ext.field.Spinner control in Sencha Touch 2

The following example shows how you can toggle value wrapping on an Ext.field.Spinner (spinnerfield) component in Sencha Touch 2 by setting the Boolean cycle config option or by calling the setCycle() method at runtime.

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.form.Panel', {
            fullscreen: true,
            items: [{
                xtype: 'toolbar',
                docked: 'top',
                items: {
                    text: 'toggle cycle',
                    handler: function () {
                        var spnnr = Ext.getCmp('spnnr');
                        spnnr.setCycle(!spnnr.getCycle());
                        spnnr.setLabel('cycle = ' + spnnr.getCycle());
                    } // handler
                } // items
            }, {
                xtype: 'fieldset',
                instructions: 'minValue = 10; maxValue = 20;',
                items: [{
                    xtype: 'spinnerfield',
                    id: 'spnnr',
                    cycle: true,
                    label: 'cycle = true',
                    minValue: 10,
                    maxValue: 20
                }] // items (fieldset)
            }] // items (formpanel)
        }); // create()
    } // launch
}); // application()
