Setting the icon alignment on an Ext.Button component using Sencha Touch 2

The following example shows how you can align the icon within an Ext.Button component using Sencha Touch 2 by setting the iconAlign config to "top", "bottom", "left", or "right".

Tags: Sencha Touch Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.Panel', {
            fullscreen: true,
            styleHtmlContent: true,
            defaults: {
                margin: 5
            }, // defaults
            layout: {
                type: 'vbox',
                align: 'center'
            },
            items: [{
                xtype: 'button',
                text: 'iconAlign: left',
                iconAlign: 'left',
                iconCls: 'arrow_left',
                iconMask: true
            }, {
                xtype: 'button',
                text: 'iconAlign: right',
                iconAlign: 'right',
                iconCls: 'arrow_right',
                iconMask: true
            }, {
                xtype: 'button',
                text: 'iconAlign: top',
                iconAlign: 'top',
                iconCls: 'arrow_up',
                iconMask: true
            }, {
                xtype: 'button',
                text: 'iconAlign: bottom',
                iconAlign: 'bottom',
                iconCls: 'arrow_down',
                iconMask: true
            }] // items
        }); // create()
    } // launch
}); // application()
