Adding a title to an Ext.ActionSheet container in Sencha Touch 2

The following example shows how you can add a simple title to an Ext.ActionSheet container in Sencha Touch 2 by nesting an Ext.TitleBar component.

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        var actionSheet = Ext.create('Ext.ActionSheet', {
            defaults: {
                handler: function (btn, evt) {
                    actionSheet.hide();
                } // handler
            }, // defaults
            items: [{
                xtype: 'titlebar',
                title: 'What are we doing?',
                ui: 'light'
            }, {
                text: 'Rock'
            }, {
                text: 'Paper'
            }, {
                text: 'Scissors'
            }] // items
        }); // create()
        
        Ext.create('Ext.Container', {
            fullscreen: true,
            items: [{
                xtype: 'titlebar',
                items: [{
                    xtype: 'button',
                    text: 'Show ActionSheet',
                    ui: 'confirm',
                    handler: function () {
                        actionSheet.show();
                    } // handler
                }] // items
            },
            actionSheet] // items
        }); // create()
    } // launch
}); // application()
