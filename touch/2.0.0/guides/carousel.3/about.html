Using Carousels in Sencha Touch 2

Carousels allow you to swipe between screens and work well on tablets and phone screens.

Tags: Sencha Touch Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.Carousel', {
            fullscreen: true,
            items: [{
                xtype: 'list',
                items: {
                    xtype: 'toolbar',
                    docked: 'top',
                    title: 'Sencha Touch Team'
                },
                store: {
                    fields: ['name'],
                    data: [{
                        name: 'Rob'
                    }, {
                        name: 'Ed'
                    }, {
                        name: 'Jacky'
                    }, {
                        name: 'Jamie'
                    }, {
                        name: 'Tommy'
                    }, {
                        name: 'Abe'
                    }]
                },
                itemTpl: '{name}'
            }, {
                xtype: 'fieldset',
                items: [{
                    xtype: 'toolbar',
                    docked: 'top',
                    title: 'Login'
                }, {
                    xtype: 'textfield',
                    label: 'Name'
                }, {
                    xtype: 'passwordfield',
                    label: 'Password'
                }]
            }]
        });
    } // launch
}); // application()
